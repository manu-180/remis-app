# Design System (Tanda 1B) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `packages/design-system` — the production-grade, multi-platform design token system for the remis app, exporting CSS variables + TypeScript constants (for Next.js/Tailwind) and Dart constants + ThemeData + Flutter primitive widgets.

**Architecture:** `tokens.json` is the single source of truth in Style Dictionary v4 format. `build/build.ts` (tsx) runs Style Dictionary to emit `src/css/tokens.css` (light `:root` + dark `[data-theme="dark"]`), and also directly generates `src/tokens.ts` and `src/tailwind-preset.ts` by importing `tokens.json`. `build/flutter_build.dart` reads `tokens.json` directly and emits `flutter/lib/tokens.dart`. Flutter theme and widgets consume `RTokens` constants only — zero hardcoded hex.

**Tech Stack:** Style Dictionary ^4.2.0, tsx ^4.19, TypeScript ^5.7, culori ^4 (OKLCH validation), pnpm 9.15, Dart SDK ^3.5, Flutter ^3.27, google_fonts ^6.2.1

**File ownership:** Only write inside `packages/design-system/**`. Never touch other packages.

function remisName(token: TransformedToken): string {
  const path = [...token.path];
  const last = path[path.length - 1];
  if (last === 'light' || last === 'dark') path.pop();
  const [cat, ...rest] = path;

  switch (cat) {

      const [sub, ...cr] = rest;
      switch (sub) {
        case 'brand':    return 'brand-' + cr.join('-').replace(/_/g, '-');
        case 'neutral':  return 'neutral-' + cr[0];
        case 'semantic': {

        }
        case 'driver_status': return 'driver-status-' + cr[0]!.replace(/_/g, '-');
        case 'ride_status':   return 'ride-status-' + cr.join('-').replace(/_/g, '-');

void _writeColors(StringBuffer buf, Map<String, dynamic> colors) {
  for (final cat in colors.entries) {
    buf.writeln('  // Color: ${cat.key}');
    _walkColorNode(buf, cat.key, cat.value as Map<String, dynamic>);

void _walkColorNode(StringBuffer buf, String prefix, Map<String, dynamic> node) {
  if (node.containsKey('value')) {
    // leaf — should not happen at this level for themed tokens
    return;
  }
  for (final entry in node.entries) {
    final child = entry.value;
    if (child is! Map<String, dynamic>) continue;

    if (child.containsKey('light') && child.containsKey('dark')) {
      // themed token pair
      final lightVal = ((child['light'] as Map)['value'] as String);
      final darkVal  = ((child['dark']  as Map)['value'] as String);
      final lightName = _colorVarName('${prefix}_${entry.key}', 'light');
      final darkName  = _colorVarName('${prefix}_${entry.key}', 'dark');
      buf.writeln('  static const $lightName = ${_hexToColor(lightVal)};');
      buf.writeln('  static const $darkName  = ${_hexToColor(darkVal)};');
    } else if (child.containsKey('value')) {
      // non-themed color leaf
      final val = child['value'] as String;
      final name = _colorVarName('${prefix}_${entry.key}', null);
      buf.writeln('  static const $name = ${_hexToColor(val)};');
    } else {
      // deeper nesting
      _walkColorNode(buf, '${prefix}_${entry.key}', child);
    }
  }
}

// ── Helpers ───────────────────────────────────────────────────────────────────

String _hexToColor(String hex) {
  final clean = hex.replaceFirst('#', '');
  return 'Color(0xFF${clean.toUpperCase()})';
}

double _parsePx(String v) => double.parse(v.replaceAll('px', ''));

int _parseMs(String v) => int.parse(v.replaceAll('ms', ''));

String _capitalize(String s) => s[0].toUpperCase() + s.substring(1);

String _camel(String prefix, String key) {
  final parts = key.split(RegExp(r'[-_]'));
  final rest = parts.map((p) => p.isEmpty ? '' : p[0].toUpperCase() + p.substring(1)).join('');
  return prefix + rest;
}

String _colorVarName(String raw, String? mode) {
  // e.g. raw='brand_primary', mode='light' → brandPrimaryLight
  final parts = raw.split(RegExp(r'[-_]'));
  final camelParts = parts.map((p) {
    if (p.isEmpty) return '';
    return p[0].toUpperCase() + p.substring(1);
  }).toList();
  final base = camelParts[0][0].toLowerCase() + camelParts[0].substring(1) + camelParts.skip(1).join('');
  if (mode == null) return base;
  return base + _capitalize(mode);
}

// 3-stop bottom sheet: collapsed (120), half (360), full
class RBottomSheet extends StatelessWidget {
  final Widget collapsed;
  final Widget half;
  final Widget full;
  final double collapsedSize;
  final double halfSize;

  const RBottomSheet({
    super.key,
    required this.collapsed,
    required this.half,
    required this.full,
    this.collapsedSize = 120,
    this.halfSize = 360,
  });

  @override
  Widget build(BuildContext context) {
    final isDark = Theme.of(context).brightness == Brightness.dark;
    final bg = isDark ? RTokens.neutral100Dark : RTokens.neutral0Light;
    final handle = isDark ? RTokens.neutral300Dark : RTokens.neutral300Light;
    final screenH = MediaQuery.sizeOf(context).height;

    return DraggableScrollableSheet(
      initialChildSize: collapsedSize / screenH,
      minChildSize: collapsedSize / screenH,
      maxChildSize: 1.0,
      snap: true,
      snapSizes: [collapsedSize / screenH, halfSize / screenH, 1.0],
      builder: (context, scrollController) {
        // Determine which stop we're at to show the right content
        final screenH = MediaQuery.sizeOf(context).height;
        final currentChildSize = scrollController.hasClients && scrollController.position.hasContentDimensions
            ? scrollController.position.extentTotal / screenH
            : collapsedSize / screenH;
        final currentStop = _currentStop(currentChildSize: currentChildSize, screenH: screenH);

        return Container(
          decoration: BoxDecoration(
            color: bg,
            borderRadius: const BorderRadius.vertical(top: Radius.circular(16)),
            boxShadow: [BoxShadow(color: Colors.black.withOpacity(0.12), blurRadius: 24, offset: const Offset(0, -4))],
          ),
          child: SingleChildScrollView(
            controller: scrollController,
            child: Column(
              children: [
                SizedBox(height: RTokens.space12),
                // Drag handle — 44×4 per spec
                Container(
                  width: 44, height: 4,
                  decoration: BoxDecoration(
                    color: handle,
                    borderRadius: BorderRadius.circular(2),
                  ),
                ),
                SizedBox(height: RTokens.space12),
                // Collapsed content is always shown
                collapsed,
                // Half and full content shown when expanded past collapsed stop
                if (currentStop >= 1) half,
                if (currentStop >= 2) full,
              ],
            ),
          ),
        );
      },
    );
  }

  int _currentStop({required double currentChildSize, required double screenH}) {
    if (currentChildSize <= (collapsedSize / screenH) + 0.05) return 0;
    if (currentChildSize <= (halfSize / screenH) + 0.05) return 1;
    return 2;
  }
}

class _ToastPositioned extends StatefulWidget {

  final VoidCallback onDismiss;

  State<_ToastPositioned> createState() => _ToastPositionedState();

class _ToastPositionedState extends State<_ToastPositioned>
    with SingleTickerProviderStateMixin {

  late final Animation<double> _fade;

  void initState() {
    super.initState();
    _c = AnimationController(vsync: this, duration: const Duration(milliseconds: 240));
    _fade = CurvedAnimation(parent: _c, curve: Curves.easeOut);

        child: SlideTransition(
          position: Tween<Offset>(begin: const Offset(0, -0.3), end: Offset.zero)
              .animate(CurvedAnimation(parent: _c, curve: Curves.easeOut)),
          child: widget.toast,

      ),
    );
  }

# @remis/design-system

Design tokens and UI primitives for the remis app. Source of truth: `tokens.json`.

## Usage in Next.js (Tailwind v4)

\`\`\`ts
// tailwind.config.ts
import remisPreset from '@remis/design-system/tailwind';

@import '@remis/design-system/css';

\`\`\`yaml
# pubspec.yaml
dependencies:
  remis_design_system:
    path: ../../packages/design-system/flutter
\`\`\`

\`\`\`dart
import 'package:remis_design_system/remis_design_system.dart';

MaterialApp(

# Flutter (tokens.dart)
cd packages/design-system && dart run build/flutter_build.dart
\`\`\`

- **Never use hex directly in UI files.** Always reference a token.
- **Never invent intermediate values** not in `tokens.json`. If a value is missing, add it to `tokens.json` and regenerate.
